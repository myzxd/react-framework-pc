/**
 * 异常账户管理相关model
 *
 * @module model/account/exception
 */
/* eslint no-underscore-dangle: ["error", { "allow": ["_meta"] }]*/
import is from 'is_js';
import { message } from 'antd';
import moment from 'moment';

import { ResponseMeta, RequestMeta } from '../../application/object';
import { fetchAccountException } from '../../services/system';
import { authorize } from '../../application';
import accountException from '../../application/utils/accountException';
import Operate from '../../application/define/operate';
import { AbnormalAccountListItem } from '../../application/object/salary/test';
import { fetchExceptionAccounts, operateExceptionAccounts } from '../../services/account';

export default {
  /**
   * 命名空间
   * @default
   */
  namespace: 'accountException',
  /**
   * 状态树
   * @prop {object} exceptionAccounts 账户列表数据
   */
  state: {
    exceptionAccounts: {},       // 账户列表数据
  },

  /**
   * @namespace account/exception/subscriptions
   */
  subscriptions: {
    setup({ dispatch, history }) {
      history.listen(() => {
        // 如果账号处于未登录，不进行处理
        if (authorize.isLogin() === false) {
          return;
        }
        // 超管不调用异常账号处理;
        if (authorize.account.role.id === 1000) {
          return;
        }
        // 检查状态，判断是否有权限查看异常的账号
        if (Operate.canOperateSystemCheckAccountException() === false) {
          return;
        }

        // 判断是否显示异常页面
        if (accountException.isDisplayException() === true) {
          window.location.href = '/#/System/Account/Exception';
          return;
        }

        // 判断是否检测账户异常
        if (accountException.isCheckException() === true) {
          dispatch({ type: 'checkAccountExcepition', payload: { account_id: authorize.account.id } });
        }
      });
    },
  },

  /**
   * @namespace account/exception/effects
   */
  effects: {
    /**
     * 检查是否有异常员工数据
     * @todo 接口需升级优化
     * @memberof module:model/account/exception~account/exception/effects
     */
    * checkAccountExcepition({ payload }, { call }) {
      const result = yield call(fetchAccountException, payload);
      if (result === undefined) {
        return;
      }

      if (result.flag !== true) {
        // 记录当前请求的时间戳
        accountException.exceptionChecked(false);
        return;
      }

      // 记录当前请求的时间戳
      accountException.exceptionChecked(true);

      // 是否跳转到异常界面
      if (accountException.isDisplayException() === true) {
        window.location.href = '/#/System/Account/Exception';
      }
    },

    /**
     * 获取异常账号列表
     * @param {array} suppliers 供应商
     * @param {array} platforms 平台
     * @param {array} cities 城市
     * @param {array} districts 商圈
     * @param {string} name 姓名
     * @param {string} courierId 三方id
     * @param {object} remindDate 发现时间
     * @memberof module:model/account/exception~account/exception/effects
     */
    *fetchExceptionAccounts({ payload = {} }, { call, put }) {
      // 默认参数
      const params = {
        _meta: RequestMeta.mapper(payload),
      };
      // 供应商
      if (is.existy(payload.suppliers) && is.not.empty(payload.suppliers) && is.array(payload.suppliers)) {
        params.supplier_id = payload.suppliers;
      }
      // 平台
      if (is.existy(payload.platforms) && is.not.empty(payload.platforms) && is.array(payload.platforms)) {
        params.platform_code = payload.platforms;
      }
      // 城市
      if (is.existy(payload.cities) && is.not.empty(payload.cities) && is.array(payload.cities)) {
        params.city_code = payload.cities;
      }
      // 商圈
      if (is.existy(payload.districts) && is.not.empty(payload.districts) && is.array(payload.districts)) {
        params.biz_district_id = payload.districts;
      }
      // 姓名
      if (is.existy(payload.name) && is.not.empty(payload.name)) {
        params.name = payload.name;
      }
      // 三方ID
      if (is.existy(payload.courierId) && is.not.empty(payload.courierId)) {
        params.courier_id = payload.courierId;
      }
      // 发现时间
      if (is.existy(payload.remindDate) && is.not.empty(payload.remindDate) && is.array(payload.remindDate)) {
        const startDate = payload.remindDate[0] ? moment(payload.remindDate[0]).format('YYYY-MM-DD') : '';
        const endDate = payload.remindDate[1] ? moment(payload.remindDate[1]).format('YYYY-MM-DD') : '';
        params.created_at = [startDate, endDate];
      }

      // 请求服务器
      const result = yield call(fetchExceptionAccounts, params);
      // 判断数据是否为空
      if (is.existy(result)) {
        yield put({ type: 'reduceExceptionAccounts', payload: result });
      }
    },

    /**
     * 操作异常账号列表
     * @param {array} selectedDataIds 操作数据id数组
     * @memberof module:model/account/exception~account/exception/effects
     */
    *operateExceptionAccounts({ payload = {} }, { call }) {
      const params = {
        abnormal_account_ids: [],  // 向后台发送操作数据id数组
      };
      if (is.existy(payload.selectedDataIds) && is.not.empty(payload.selectedDataIds) && is.array(payload.selectedDataIds)) {
        params.abnormal_account_ids = payload.selectedDataIds;
      }
      // 请求服务器
      const result = yield call(operateExceptionAccounts, params);
      // 判断数据是否为空
      if (is.existy(result)) {
        // 操作成功的回调函数
        if (result.ok) {
          message.success('操作成功');
          if (payload.onSuccessCallback) {
            payload.onSuccessCallback();
          }
        } else {
          message.error('操作失败');
        }
      }
    },
  },

  /**
   * @namespace account/exception/reducers
   */
  reducers: {
    /**
     * 异常账号
     * @return {object} 更新 exceptionAccounts
     * @memberof module:model/account/exception~account/exception/effects
     */
    reduceExceptionAccounts(state, action) {
      const exceptionAccounts = {
        meta: ResponseMeta.mapper(action.payload._meta) || {},
        data: AbnormalAccountListItem.mapperEach(action.payload.data, AbnormalAccountListItem) || [],
      };
      return { ...state, exceptionAccounts };
    },
  },
};
